			not = { has_global_flag = crDFR }
			not = { war_with = sov }
			not = { war_with = ENG }
			not = { war_with = FRA }			
			not = { war_with = JAP }
			not = { war_with = ITA }			
			not = { exists = FRG }
			year = 1946
			not = { year = 1960 }
			is_subject = no 
	}
	mean_time_to_happen = { months = 6 }

	title = "The Allies are considering the creation of German state"
	desc = "creacion_DFR_desc"
	picture = "ONU"

	option = {
			name = "Sí"
	    ai_chance = { factor = 70 }
            set_global_flag = crDFRUS


	}
	option = {
			name = "NO"
		  ai_chance = { factor = 30 }
		  			random = {
				chance = 5			
		SOV = { war_with = USA }
		SOV = { war_with = FRA }
		SOV = { war_with = ENG } 
        set_global_flag = iniIIIWW		
		}
		
		       
                   clr_global_flag = CRDFRFR
